public void medusaTest(Stage primaryStage) {
        Gauge gauge = GaugeBuilder.create()
                .skinType(SkinType.MODERN)
                .decimals(0)
                .maxValue(20)
                .title("Code Time")
                .subTitle("in 20 ms")
                .subTitleColor(Color.GRAY)
                .tickLabelColor(Color.CADETBLUE)
                .unit(" ms")
                .sections(
                        new Section(0, 5, Color.LIGHTGREEN),
                        new Section(5, 10, Color.GREENYELLOW),
                        new Section(10, 15, Color.ORANGE),
                        new Section(15, 20, Color.RED))
                .gradientBarEnabled(true)
                .barEffectEnabled(true)
                .gradientBarStops(new Stop(0, Color.CORNFLOWERBLUE), new Stop(1, Color.RED))
                .sectionsVisible(true)
                .build();
//        gauge.setSections(
//                new Section(0, 5, "Efficient", Color.LIGHTGREEN),
//                new Section(5, 10, "Half capacity", Color.GREEN),
//                new Section(10, 15, "Beyond half capacity", Color.YELLOW),
//                new Section(15, 20, "Less than full capacity", Color.ORANGE),
//                new Section(20, 25, "Capacity Exceeded", Color.ORANGERED),
//                new Section(25, 40, "Thrashing", Color.RED)
//                          );
//        gauge.setHighlightSections(true);
//        gauge.setSectionTextVisible(true);
//        gauge.setSectionsVisible(true);
//        gauge.setNeedleType(Gauge.NeedleType.STANDARD);
        ComboBox<SkinType> skinType = new ComboBox<>(FXCollections.observableArrayList(SkinType.values()));
        skinType.setOnAction(ae -> {
            gauge.setSkinType(skinType.getValue());
        });
        Spinner value = new Spinner(0, 21, 4);
        gauge.valueProperty().bind(value.valueProperty());
        final VBox vBox = new VBox(gauge, value, skinType);
        vBox.setMaxHeight(Double.MAX_VALUE);
        vBox.setMaxWidth(Double.MAX_VALUE);
        vBox.setFillWidth(true);
        vBox.setVgrow(value, Priority.ALWAYS);
        vBox.setSpacing(10);
        Scene scene = new Scene(new StackPane(vBox));
        primaryStage.setScene(scene);
        primaryStage.show();
    }
